@startuml
autonumber
participant ":api/reservasController" as RC
participant ":api/reservaDTO" as R
participant "API.QRCode" as U

activate RC

activate R
R --> RC : (reservaDTO)
deactivate R

activate U

deactivate U

deactivate RC
